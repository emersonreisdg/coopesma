Limpando tabela do db.sqlite3 e povoando-a com novos dados armazenados em arquivo csv utf8
Usando o shell do Django

    python manage.py shell

    # Limpar a tabela
    PrevisaoOrcamentaria.objects.all().delete()

    import csv
    from datetime import datetime
    from your_app.models import PrevisaoOrcamentaria

    # Caminho para o arquivo CSV
    csv_file_path = 'caminho/para/seu/arquivo/orcamento2025.csv'

    # Função para converter valores monetários
    def parse_valor(valor_str):
        return float(valor_str.replace('R$', '').replace('.', '').replace(',', '.').strip())

    # Abrir e ler o arquivo CSV
    with open(csv_file_path, newline='', encoding='utf-8') as csvfile:
        reader = csv.DictReader(csvfile, delimiter=';')
        for row in reader:
            data = datetime.strptime(row['Data'], '%d/%m/%Y')
            valor = parse_valor(row['Valor'])
            previsao = PrevisaoOrcamentaria(
                data=data,
                tipo=row['Tipo'],
                subtipo=row['Subtipo'],
                categoria=row['Categoria'],
                item=row['Item'],
                valor=valor,
                observacao=row['Observação']
            )
            previsao.save()
